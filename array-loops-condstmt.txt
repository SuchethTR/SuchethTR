        nr = Convert.ToInt32(Console.ReadLine());

                for (j = 2; j < n - i; j++)
                {
                    if ((n - i) % j == 0)
                    { n2 = 0; j = n - i; }
                }
                if (n1 == 1 && n2 == 1)
                {
                    Console.Write("{0} =  {1} + {2}  \n", n, i, n - i);
                    n3 = 1;
                }
            }
            if (n3 == 0)
            { Console.Write("\n{0} can not be expressed as sum of two prime numbers.\n", n); }
        }
    }
